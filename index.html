#precision function 
precision <- function(x.pred,x.obs){
  mx <- table(x.pred,x.obs)
  z=0
  for(i in 1:length(unique(x.obs))){
    z <- z+mx[i,i]/sum(mx[i,])
  }
  return(z/i)
}

The interest is predicting a logged value for triceps skinfold thickness given an age.

(a)

As it is seen in the above scatter plot, logged values for triceps skinfold thickness are not linear with ages. A linear regression using a B-spline of the order 3 is considered and, a range of possible number of knots is [0, 20].

Find the number of knots minimizing the cross validation error using 10 folders.

set.seed(1234)

n.knots=c(0:20)
cv.triceps=rep(0,length(n.knots))
ii=rep(c(1:10),90); ii=ii[1:892]

for(j in 1:length(n.knots)){ for(i in 1:10){ 
  t.bs=predict(lm(log(triceps)~bs(age,df=n.knots[j]+3,Boundary.knots=c(0.2,52)),data=triceps[-which(ii==i),]),newdata=triceps[which(ii==i),])
  cv.triceps[j]=cv.triceps[j]+mean((log(triceps$triceps[which(ii==i)])-t.bs)^2)
}}
#no of knots minimizing the cv error
min.knots = n.knots[which.min(cv.triceps)]
