Find the number of knots minimizing the cross validation error using 10 folders.

trainRt=read.csv('Rent.csv',header = TRUE)
plot(trainRt$Year,trainRt$MRent,col="grey",main="Rent weekly ~ Year",xlab = "Year",ylab = "MedianWeeklyRent (NZD)",cex.main=1)

# vector of no of knots
year.mse.bs=v=c(0:20)

ii=rep(c(1:10),554) 
ii=ii[1:nrow(trainRt)]

for(j in 1:length(v)){
  for(i in 1:max(ii)){

    t.bs=bs(trainRt$Year[ii!=i],df=3+v[j])

    t.test.bs=predict(t.bs,trainRt$Year[ii==i])
    trainData=data.frame(Year=t.bs,MRent=trainRt$MRent[ii!=i])

    pre.t.bs=predict(lm(MRent~.,trainData),testData)

  year.mse.bs[j]=year.mse.bs[j]+mean((testData$MRent-pre.t.bs)^2)
